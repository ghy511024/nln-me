打印杨辉三角


 function getyanghui(n) {
        var tmparray = [];
        var allarray = [];
        for (var i = 0; i < n; i++) {
            var arr = []
            for (var j = 0; j < i; j++) {
                if (j == 0) {
                    arr[j] = tmparray[j]
                } else {
                    arr[j] = tmparray[j] + tmparray[j - 1];
                }
            }
            if (tmparray.length == 0) {
                arr[i] = 1;
            } else {
                arr[i] = tmparray[i - 1]
            }
            tmparray = arr;
            allarray.push(arr)
        }
        return allarray;
    }

超出部分省略号

.els {
  @include els;
}

.els {
  overflow: hidden;
  white-space: nowrap;
  text-overflow: ellipsis;
  width: 100%;
}
清除浮动

.clear {
  @include clearfix;
}

.clear {
  *zoom: 1;
}

.clear:before, .clear:after {
  content: "";
  display: table;
}

.clear:after {
  clear: both;
}
兼容的inline-block

.inblock {
  @include inblock;
}

.inblock {
  display: -moz-inline-stack;
  display: inline-block;
  vertical-align: top;
  zoom: 1;
  *display: inline;
}
绝对布局居中

.center {
  // 上下都居中
  &.all {
    @include center(all);
  }
  // 水平居中
  &.w {
    @include center(w);
  }
  //垂直居中
  &.h {
    @include center(h);
  }
}

.center.all {
  position: absolute;
  top: 50%;
  left: 50%;
  -webkit-transform: translate3d(-50%, -50%, 0);
  transform: translate3d(-50%, -50%, 0);
}

.center.w {
  position: absolute;
  left: 50%;
  -webkit-transform: translate3d(-50%, 0, 0);
  transform: translate3d(-50%, 0, 0);
}

.center.h {
  position: absolute;
  top: 50%;
  -webkit-transform: translate3d(0, -50%, 0);
  transform: translate3d(0, -50%, 0);
}
透明度

.opacity {
  @include opacity(0.5);
}

.opacity {
  filter: alpha(opacity=50);
  opacity: 0.5;
}
placeholder

.input {
  @include placeholder() {
    color: #f5f5f5;
  }
}

.input::-webkit-input-placeholder {
  color: #f5f5f5;
}

.input::-moz-placeholder {
  color: #f5f5f5;
}

.input:-ms-input-placeholder {
  color: #f5f5f5;
}
animate 

.animate {
  @include animation(slideUp 900ms ease both) {
    0% {
      transform: translate3d(0, -200px, 0);
    }
    100% {
      transform: translate3d(0, 0, 0);
    }
  }
}

.animate {
